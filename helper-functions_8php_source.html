291 /**
292  * Is the value empty?
293  *
294  * Allows you to pass a function instead of just a variable, like the empty() function insists upon (until PHP 5.5)
295  *
296  * Checks whether `false`, `null`, empty string, empty array, object with no vars defined
297  *
298  * @since 1.15.1
299  * @param mixed $value Check whether this is empty
300  * @param boolean $zero_is_empty Should the number zero be treated as an empty value?
301  * @param boolean $allow_string_booleans Whether to check if 'yes', 'true' => `true` and 'no', 'false' => `false`
302  * @return boolean True: empty; false: not empty
303  */
304 function gv_empty( $value, $zero_is_empty = true, $allow_string_booleans = true ) {
305 
306  if(
307  ! isset( $value ) // If it's not set, it's empty!
308  || false === $value
309  || null === $value
310  || '' === $value // Empty string
311  || array() === $value // Empty array
312  || ( is_object( $value ) && ! get_object_vars( $value ) ) // Empty object
313  ) {
314  return true;
315  }
316 
317  if( is_string( $value ) && $allow_string_booleans ) {
318 
319  $value = trim( $value );
320  $value = strtolower( $value );
321 
322  if ( in_array( $value, array( 'yes', 'true' ), true ) ) {
323  $value = true;
324  } else if( in_array( $value, array( 'no', 'false' ), true ) ) {
325  $value = false;
326  }
327  }
328 
329  // If zero isn't empty, then if $value is a number and it's empty, it's zero. Thus, return false.
330  if( ! $zero_is_empty && is_numeric( $value ) && empty( $value ) ) {
331  return false;
332  }
333 
334  return empty( $value );
335 }
