866 /**
867  * Get installed translations.
868  *
869  * Looks in the wp-content/languages directory for translations of
870  * plugins or themes.
871  *
872  * @since 3.7.0
873  *
874  * @param string $type What to search for. Accepts 'plugins', 'themes', 'core'.
875  * @return array Array of language data.
876  */
877 function wp_get_installed_translations( $type ) {
878  if ( $type !== 'themes' && $type !== 'plugins' && $type !== 'core' )
879  return array();
880 
881  $dir = 'core' === $type ? '' : "/$type";
882 
883  if ( ! is_dir( WP_LANG_DIR ) )
884  return array();
885 
886  if ( $dir && ! is_dir( WP_LANG_DIR . $dir ) )
887  return array();
888 
889  $files = scandir( WP_LANG_DIR . $dir );
890  if ( ! $files )
891  return array();
892 
893  $language_data = array();
894 
895  foreach ( $files as $file ) {
896  if ( '.' === $file[0] || is_dir( WP_LANG_DIR . "$dir/$file" ) ) {
897  continue;
898  }
899  if ( substr( $file, -3 ) !== '.po' ) {
900  continue;
901  }
902  if ( ! preg_match( '/(?:(.+)-)?([a-z]{2,3}(?:_[A-Z]{2})?(?:_[a-z0-9]+)?).po/', $file, $match ) ) {
903  continue;
904  }
905  if ( ! in_array( substr( $file, 0, -3 ) . '.mo', $files ) ) {
906  continue;
907  }
908 
909  list( , $textdomain, $language ) = $match;
910  if ( '' === $textdomain ) {
911  $textdomain = 'default';
912  }
913  $language_data[ $textdomain ][ $language ] = wp_get_pomo_file_data( WP_LANG_DIR . "$dir/$file" );
914  }
915  return $language_data;
916 }
