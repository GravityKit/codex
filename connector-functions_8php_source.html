407  * @see GravityView_frontend::add_scripts_and_styles()
408  * @since 1.15

410  * @param array $view_fields GravityView fields array

412 function gravityview_view_has_single_checkbox_or_radio( $form, $view_fields ) {
413 
414  if( class_exists('GFFormsModel') && $form_fields = GFFormsModel::get_fields_by_type( $form, array( 'checkbox', 'radio' ) ) ) {
415 
416  /** @var GF_Field_Radio|GF_Field_Checkbox $form_field */
417  foreach( $form_fields as $form_field ) {
418  $field_id = $form_field->id;
419  foreach( $view_fields as $zone ) {
420 
421  // ACF compatibility; ACF-added fields aren't arrays
422  if ( ! is_array( $zone ) ) { continue; }
423 
424  foreach( $zone as $field ) {
425  // If it's an input, not the parent and the parent ID matches a checkbox or radio
426  if( ( strpos( $field['id'], '.' ) > 0 ) && floor( $field['id'] ) === floor( $field_id ) ) {
427  return true;
428  }
429  }
430  }
431  }
432  }
