263 /**
264  * Theme function to get a GravityView view
265  *
266  * @access public
267  * @param string $view_id (default: '')
268  * @param array $atts (default: array())
269  * @return string HTML of the output. Empty string if $view_id is empty.
270  */
271 function get_gravityview( $view_id = '', $atts = array() ) {
272  if( !empty( $view_id ) ) {
273  $atts['id'] = $view_id;
274  $args = wp_parse_args( $atts, GravityView_View_Data::get_default_args() );
275  $GravityView_frontend = GravityView_frontend::getInstance();
276  $GravityView_frontend->setGvOutputData( GravityView_View_Data::getInstance( $view_id ) );
277  $GravityView_frontend->set_context_view_id( $view_id );
278  $GravityView_frontend->set_entry_data();
279  return $GravityView_frontend->render_view( $args );
280  }
281  return '';
282 }
