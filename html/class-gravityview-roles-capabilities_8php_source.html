318  /**
319  * Check whether the current user has a capability
320  *
321  * @since 1.15
322  *
323  * @see WP_User::user_has_cap()
324  * @see https://codex.wordpress.org/Plugin_API/Filter_Reference/user_has_cap You can filter permissions based on entry/View/form ID using `user_has_cap` filter
325  *
326  * @see GFCommon::current_user_can_any
327  * @uses GFCommon::current_user_can_any
328  *
329  * @param string|array $caps_to_check Single capability or array of capabilities
330  * @param int|null $object_id (optional) Parameter can be used to check for capabilities against a specific object, such as a post or us
331  * @param int|null $user_id (optional) Check the capabilities for a user who is not necessarily the currently logged-in user
332  *
333  * @return bool True: user has at least one passed capability; False: user does not have any defined capabilities
334  */
335  public static function has_cap( $caps_to_check = '', $object_id = null, $user_id = null ) {
336 
337  /**
338  * @filter `gravityview/capabilities/allow_logged_out` Shall we allow a cap check for non-logged in users? USE WITH CAUTION!
339  *
340  * WARNING: This allows anyone to edit and delete entries, add notes, delete notes, etc!
341  *
342  * If you use this filter, at least check against certain capabilities and $object_ids.
343  *
344  * There are use-cases, albeit strange ones, where we'd like to check and override capabilities for
345  * for a non-logged in user.
346  *
347  * Examples, you ask? https://github.com/gravityview/GravityView/issues/826
348  *
349  * @param boolean $allow_logged_out Allow the capability check or bail without even checking. Default: false. Do not allow. Do not pass Go. Do not collect $200.
350  * @param string|array $caps_to_check Single capability or array of capabilities to check against
351  * @param int|null $object_id (optional) Parameter can be used to check for capabilities against a specific object, such as a post or us.
352  * @param int|null $user_id (optional) Check the capabilities for a user who is not necessarily the currently logged-in user.
353  */
354  $allow_logged_out = apply_filters( 'gravityview/capabilities/allow_logged_out', false, $caps_to_check, $object_id, $user_id );
355 
356  if ( true === $allow_logged_out ) {
357 
358  $all_caps = self::all_caps('editor');
359 
360  if( array_intersect( $all_caps, (array) $caps_to_check ) ) {
361  return true;
362  }
363  }
364 
365  /**
366  * We bail with a negative response without even checking if:
367  *
368  * 1. The current user is not logged in and non-logged in users are considered unprivileged (@see `gravityview/capabilities/allow_logged_out` filter).
369  * 2. If the caps to check are empty.
370  */
371  if ( ( ! is_user_logged_in() && ! $allow_logged_out ) || empty( $caps_to_check ) ) {
372  return false;
373  }
374 
375  $has_cap = false;
376 
377  // Add full access caps for GV & GF
378  $caps_to_check = self::maybe_add_full_access_caps( $caps_to_check );
379 
380  foreach ( $caps_to_check as $cap ) {
381  if( ! is_null( $object_id ) ) {
382  $has_cap = $user_id ? user_can( $user_id, $cap, $object_id ) : current_user_can( $cap, $object_id );
383  } else {
384  $has_cap = $user_id ? user_can( $user_id, $cap ) : current_user_can( $cap );
385  }
386  // At the first successful response, stop checking
387  if( $has_cap ) {
388  break;
389  }
390  }
391 
392  return $has_cap;
393  }
