302  *
303  * 1. Check referrer validity
304  * 2. Make sure there's an entry with the slug of $_GET['entry_id']

306  * 4. Remove `action=delete` from the URL
307  * 5. Redirect to the page using `wp_safe_redirect()`

311  * @return void
312  */
313  function process_delete() {
314

317

322 
323  // Get the entry slug
