377  * Add dependencies
378  *
379  * @param [type] $registered [description]
380  * @param [type] $scripts [description]
381  */
382  private function add_script_dependencies($registered, $scripts) {
383 
384  //gets all dependent scripts linked to the $scripts array passed
385  do {
386  $dependents = array();
387  foreach ( $scripts as $script ) {
388  $deps = isset( $registered[ $script ] ) && is_array( $registered[ $script ]->deps ) ? $registered[ $script ]->deps : array();
