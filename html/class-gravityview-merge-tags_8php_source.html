157

159  preg_match_all( "/\{created_by:(.*?)\}/", $text, $matches, PREG_SET_ORDER );
160 
161  // If there are no matches OR the Entry `created_by` isn't set or is 0 (no user)
162  if( empty( $matches ) || empty( $entry['created_by'] ) ) {
163  return $text;
164  }
165 
166  // Get the creator of the entry
167  $entry_creator = new WP_User( $entry['created_by'] );
168 
169  foreach ( $matches as $match ) {

172  $property = $match[1];
173 
174  switch( $property ) {
175  /** @since 1.13.2 */
176  case 'roles':
177  $value = implode( ', ', $entry_creator->roles );
178  break;
179  default:

181  }
