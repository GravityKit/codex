172  * - name - name of the function

174  * - line - line number
175  */
176  public function find_function_calls( $function_names, $code ) {
177  $tokens = token_get_all( $code );
178  $function_calls = array();

180  $in_func = false;
181  foreach( $tokens as $token ) {
182  $id = $text = null;
183  if ( is_array( $token ) ) {
184  list( $id, $text, $line ) = $token;
185  }
186  if ( T_WHITESPACE == $id ) {
187  continue;
188  }
189  if ( T_STRING == $id && in_array( $text, $function_names ) && ! $in_func ) {
190  $in_func = true;
191  $paren_level = -1;
192  $args = array();
193  $func_name = $text;
194  $func_line = $line;
195  $func_comment = $latest_comment? $latest_comment : '';

198  $latest_comment = false;
199  continue;
200  }
