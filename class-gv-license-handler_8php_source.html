271  // Empty is returned when there's an error.
272  if ( empty( $license_data ) ) {
273  if ( $is_ajax ) {

275  } else { // Non-ajax call
276  return json_encode( array() );

278  }
279 
280  $license_data->message = $this->get_license_message( $license_data );
281 
282  $json = json_encode( $license_data );
283 
284  $update_license = ( !isset( $data['update'] ) || !empty( $data['update'] ) );
285 
286  $is_check_action_button = ( 'check_license' === $data['edd_action'] && defined('DOING_AJAX') && DOING_AJAX );
287 
288  // Failed is the response from trying to de-activate a license and it didn't work.
289  // This likely happened because people entered in a different key and clicked "Deactivate",
290  // meaning to deactivate the original key. We don't want to save this response, since it is
291  // most likely a mistake.
292  if ( $license_data->license !== 'failed' && !$is_check_action_button && $update_license ) {
293 
294  if( !empty( $data['field_id'] ) ) {
295  set_transient( 'gravityview_' . esc_attr( $data['field_id'] ) . '_valid', $license_data, DAY_IN_SECONDS );
296  }
297 
298  $this->license_call_update_settings( $license_data, $data );
299 
300  }
301 
302  if ( $is_ajax ) {
303  exit( $json );
