203  function update_search_on_views() {
204

226

230 
231  if( is_null( $search_fields ) ) {
232  $search_fields = $this->get_search_fields( $view->ID );
233  }
234

236  // [search_free] => 1
237  // [search_date] => 1

240  $search_generic[] = array( 'field' => 'search_all', 'input' => 'input_text' );

244  }

247
