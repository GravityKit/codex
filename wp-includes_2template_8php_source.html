425  *
426  * The attachment path first checks if the first part of the mime type exists.
427  * The second check is for the second part of the mime type. The last check is
428  * for both types separated by an underscore. If neither are found then the file
429  * 'attachment.php' is checked and returned.
430  *
431  * Some examples for the 'text/plain' mime type are 'text.php', 'plain.php', and
432  * finally 'text-plain.php'.
433  *
434  * The template path is filterable via the dynamic {@see '$type_template'} hook,
435  * e.g. 'attachment_template'.
436  *
437  * @since 2.0.0

440  *

443  * @return string Full path to attachment template file.

447

449

452  list( $type, $subtype ) = explode( '/', $attachment->post_mime_type );

468 /**
469  * Retrieve the name of the highest priority template file that exists.
470  *
471  * Searches in the STYLESHEETPATH before TEMPLATEPATH so that themes which
472  * inherit from a parent theme can just overload one file.
473  *
474  * @since 2.7.0
475  *
476  * @param string|array $template_names Template file(s) to search for, in order.
477  * @param bool $load If true the template file will be loaded if it is found.
478  * @param bool $require_once Whether to require_once or require. Default true. Has no effect if $load is false.
479  * @return string The template filename if one is located.
480  */
481 function locate_template($template_names, $load = false, $require_once = true ) {
482  $located = '';
483  foreach ( (array) $template_names as $template_name ) {
484  if ( !$template_name )
485  continue;
486  if ( file_exists(STYLESHEETPATH . '/' . $template_name)) {
487  $located = STYLESHEETPATH . '/' . $template_name;
488  break;
489  } elseif ( file_exists(TEMPLATEPATH . '/' . $template_name) ) {
490  $located = TEMPLATEPATH . '/' . $template_name;
491  break;
492  }
493  }
494 
495  if ( $load && '' != $located )
496  load_template( $located, $require_once );
497 
498  return $located;
499 }
