85  // Encoding list
86  // Listed fully below. Add in this format to get new phases, which will
87  // be used automatically by the enkode function.
88  $this->enk_dec_reverse = <<<EOT
89 kode=kode.split('').reverse().join('')
90 EOT;
91  $this->enk_dec_num = <<<EOT
92 kode=kode.split(' ');x='';for(i=0;i<kode.length;i++){x+=String.fromCharCode(parseInt(kode[i])-3)}kode=x
93 EOT;
94  $this->enk_dec_swap = <<<EOT
95 x='';for(i=0;i<(kode.length-1);i+=2){x+=kode.charAt(i+1)+kode.charAt(i)}kode=x+(i<kode.length?kode.charAt(kode.length-1):'')
96 EOT;

177  /**
178  * Enkode
179  *
180  * Encodes a string to be view-time written by obfuscated Javascript.
181  * The max passes parameter is a tight bound on the number of encodings
182  * perormed. The max length paramater is a loose bound on the length of
183  * the generated Javascript. Setting it to 0 will use a single pass of
184  * enkEncNum.
185  *
186  * The function works by selecting encodings at random from the array
187  * enkodings, applying them to the given string, and then producing
188  * Javascript to decode. The Javascript works by recursive evaluation,
189  * which should be nasty enough to stop anything but the most determined
190  * spambots.
191  *
192  * The text parameter, if set, overrides the user-settable option
193  * enk_msg. This is the message overwritten by the JavaScript; if a
194  * browser doesn't support JavaScript, this message will be shown to the
195  * user.
196  */
197  public function enkode($content, $text = NULL) {
198  $max_passes = $this->max_passes;
199  $max_length = $this->max_length;
200 
201  // Our base case -- we'll eventually evaluate this code.
202  // Note that we're using innerHTML() since document.write() fails on
203  // pages loaded using AJAX.
204  $kode = "document.getElementById('ENKODER_ID').outerHTML=\"" .
205  addcslashes($content,"\\\'\"&\n\r<>") .
206  "\";";
207 
208  $max_length = max($max_length, strlen($kode) + $this->min_length + 1);
209 
210  $result = "";
211 
212  // Build up as many encodings as we can.
213  for ($passes = 0; $passes < $max_passes && strlen($kode) < $max_length; $passes++) {
214  // Pick an encoding at random.
215  $idx = rand(0, count($this->enkodings) - 1);
216  $enc = $this->enkodings[$idx][0];
217  $dec = $this->enkodings[$idx][1];
218  $kode = $this->enkodePass($kode, $enc, $dec);
219  }
220 
221  // Mandatory numerical encoding, prevents catching @ signs and
222  // interpreting neighboring characters as e-mail addresses.
223  $kode = $this->enkodePass($kode, 'enkEncNum', $this->enk_dec_num);
224 
225  return $this->enkBuildJS($kode, $text);
226  }

228  /**
229  * Encode a single pass
230  *
231  * $enc is a function pointer and $dec is the Javascript.
232  */
233  private function enkodePass($kode, $enc, $dec) {
234  // First encode.
235  $kode = addslashes($this->$enc($kode));
236 
237  // Then generate encoded code with decoding afterwards.
238  $kode = "kode=\"$kode\";$dec;";
239 
240  return $kode;
241  }

243  /**
244  * Build JavaScript
245  *
246  * Generates the Javascript recursive evaluator, which is 269 characters
247  * of boilerplate code.
248  *
249  * Unfortunately, <noscript> can't be used arbitrarily in XHTML. A
250  * <span> that we immediately overwrite, serves as an ad hoc <noscript>
251  * tag.
252  */
253 
254  private function enkBuildJS($kode, $text = NULL) {
255  $clean = addslashes($kode);
256 
257  $msg = is_null($text) ? $this->enkode_msg : $text;
258 
259  $name = "enkoder_" . strval($this->enkoder_uses) . "_" . strval(rand());
260  $this->enkoder_uses += 1;
261  // Note that we decode until $kode contains "getElementById('ENKODER_ID')",
262  // at which point we replace ENKODER_ID with the ID of our span, then
263  // perform the final eval().
264  $js = <<<EOT
265 <span id="$name">$msg</span><script type="text/javascript">
266 /* <!-- */
267 function hivelogic_$name() {
268 var kode="$clean";var i,c,x;while(kode.indexOf("getElementById('ENKODER_ID')") === -1){eval(kode)};kode=kode.replace('ENKODER_ID','$name');eval(kode);
269 }
270 hivelogic_$name();
271 /* --> */
272 </script>
273 EOT;
274 
275  return $js;
276  }
