137  /**
138  * If there is a `:format` modifier in a merge tag, grab the formatting
139  *
140  * The `:format` modifier should always have the format follow it; it's the next item in the array
141  * In `foo:format:bar`, "bar" will be the returned format
142  *
143  * @since 1.16
144  *
145  * @param array $exploded Array of modifiers with a possible `format` value
146  * @param string $backup The backup value to use, if not found
147  *
148  * @return string If format is found, the passed format. Otherwise, the backup.
149  */
150  private static function get_format_from_modifiers( $exploded, $backup = '' ) {
151 
152  $return = $backup;
153 
154  $format_key_index = array_search( 'format', $exploded );
155 
156  // If there's a "format:[php date format string]" date format, grab it
157  if ( false !== $format_key_index && isset( $exploded[ $format_key_index + 1 ] ) ) {
158  // Return escaped colons placeholder
159  $return = str_replace( '|COLON|', ':', $exploded[ $format_key_index + 1 ] );
160  }
161 
162  return $return;
163  }

165  /**
166  * Allow passing variables via URL to be displayed in Merge Tags
167  *
168  * Works with `[gvlogic]`:
169  * [gvlogic if="{get:example}" is="false"]
170  * ?example=false
171  * [else]
172  * ?example wasn't "false". It's {get:example}!
173  * [/gvlogic]
174  *
175  * Supports passing arrays:
176  * URL: `example[]=Example+One&example[]=Example+(with+comma)%2C+Two`
177  * Merge Tag: `{get:example}`
178  * Output: `Example One, Example (with comma), Two`
179  *
180  * @since 1.15
181  * @param string $text Text to replace
182  * @param array $form Gravity Forms form array
183  * @param array $entry Entry array
184  * @param bool $url_encode Whether to URL-encode output
185  */
186  public static function replace_get_variables( $text, $form = array(), $entry = array(), $url_encode = false ) {
187 
188  // Is there is {get:[xyz]} merge tag?
189  preg_match_all( "/{get:(.*?)}/ism", $text, $matches, PREG_SET_ORDER );
190 
191  // If there are no matches OR the Entry `created_by` isn't set or is 0 (no user)
192  if( empty( $matches ) ) {
193  return $text;
194  }
195 
196  foreach ( $matches as $match ) {
197 
198  $full_tag = $match[0];
199  $property = $match[1];
200 
201  $value = stripslashes_deep( rgget( $property ) );
202 
203  /**
204  * @filter `gravityview/merge_tags/get/glue/` Modify the glue used to convert an array of `{get}` values from an array to string
205  * @since 1.15
206  * @param[in,out] string $glue String used to `implode()` $_GET values Default: ', '
207  * @param[in] string $property The current name of the $_GET parameter being combined
208  */
209  $glue = apply_filters( 'gravityview/merge_tags/get/glue/', ', ', $property );
210 
211  $value = is_array( $value ) ? implode( $glue, $value ) : $value;
212 
213  $value = $url_encode ? urlencode( $value ) : $value;
214 
215  /**
216  * @filter `gravityview/merge_tags/get/esc_html/{url parameter name}` Disable esc_html() from running on `{get}` merge tag
217  * By default, all values passed through URLs will be escaped for security reasons. If for some reason you want to
218  * pass HTML in the URL, for example, you will need to return false on this filter. It is strongly recommended that you do
219  * not disable this filter.
220  * @since 1.15
221  * @param bool $esc_html Whether to esc_html() the value. Default: `true`
222  */
223  $esc_html = apply_filters('gravityview/merge_tags/get/esc_html/' . $property, true );
224 
225  $value = $esc_html ? esc_html( $value ) : $value;
226 
227  /**
228  * @filter `gravityview/merge_tags/get/esc_html/{url parameter name}` Modify the value of the `{get}` replacement before being used
229  * @param[in,out] string $value Value that will replace `{get}`
230  * @param[in] string $text Text that contains `{get}` (before replacement)
231  * @param[in] array $form Gravity Forms form array
232  * @param[in] array $entry Entry array
233  */
234  $value = apply_filters('gravityview/merge_tags/get/value/' . $property, $value, $text, $form, $entry );
235 
236  $text = str_replace( $full_tag, $value, $text );
237  }
238 
239  unset( $value, $glue, $matches );
240 
241  return $text;
242  }
243 }
