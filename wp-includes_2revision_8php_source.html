582 /**
583  * Upgrade the revisions author, add the current post as a revision and set the revisions version to 1
584  *
585  * @since 3.6.0
586  * @access private
587  *
588  * @global wpdb $wpdb WordPress database abstraction object.
589  *
590  * @param WP_Post $post Post object
591  * @param array $revisions Current revisions of the post
592  * @return bool true if the revisions were upgraded, false if problems
593  */
594 function _wp_upgrade_revisions_of_post( $post, $revisions ) {
595  global $wpdb;
596 
597  // Add post option exclusively
598  $lock = "revision-upgrade-{$post->ID}";
599  $now = time();
600  $result = $wpdb->query( $wpdb->prepare( "INSERT IGNORE INTO `$wpdb->options` (`option_name`, `option_value`, `autoload`) VALUES (%s, %s, 'no') /* LOCK */", $lock, $now ) );
601  if ( ! $result ) {
602  // If we couldn't get a lock, see how old the previous lock is
603  $locked = get_option( $lock );
604  if ( ! $locked ) {
605  // Can't write to the lock, and can't read the lock.
606  // Something broken has happened
607  return false;
608  }
609 
610  if ( $locked > $now - 3600 ) {
611  // Lock is not too old: some other process may be upgrading this post. Bail.
612  return false;
613  }
614 
615  // Lock is too old - update it (below) and continue
616  }
617 
618  // If we could get a lock, re-"add" the option to fire all the correct filters.
619  update_option( $lock, $now );
620 
621  reset( $revisions );
622  $add_last = true;
623 
624  do {
625  $this_revision = current( $revisions );
626  $prev_revision = next( $revisions );
627 
628  $this_revision_version = _wp_get_post_revision_version( $this_revision );
629 
630  // Something terrible happened
631  if ( false === $this_revision_version )
632  continue;
633 
634  // 1 is the latest revision version, so we're already up to date.
635  // No need to add a copy of the post as latest revision.
636  if ( 0 < $this_revision_version ) {
637  $add_last = false;
638  continue;
639  }
640 
641  // Always update the revision version
642  $update = array(
643  'post_name' => preg_replace( '/^(\d+-(?:autosave|revision))[\d-]*$/', '$1-v1', $this_revision->post_name ),
644  );
645 
646  // If this revision is the oldest revision of the post, i.e. no $prev_revision,
647  // the correct post_author is probably $post->post_author, but that's only a good guess.
648  // Update the revision version only and Leave the author as-is.
649  if ( $prev_revision ) {
650  $prev_revision_version = _wp_get_post_revision_version( $prev_revision );
651
