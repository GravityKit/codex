991  $found = true;
992  break;
993  }
994  }
995

997  return false;
998  }
999

1001  $found = false;
1002  foreach ( $searchable_fields as $field ) {
1003  if ( $field_id == $field['field'] && $form->ID == $field['form_id'] ) {

1192 
1193  } // switch field type
