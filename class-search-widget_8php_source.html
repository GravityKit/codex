328 
329  do_action( 'gravityview_log_debug', sprintf( '%s[filter_entries] Requested $_GET: ', get_class( $this ) ), $_GET );
330 
331  if ( empty( $_GET ) || ! is_array( $_GET ) ) {
332  return $search_criteria;
333  }
334 
335  // add free search
336  if ( ! empty( $_GET['gv_search'] ) ) {
337 
338  // Search for a piece
339  $words = explode( ' ', stripslashes_deep( urldecode( $_GET['gv_search'] ) ) );
340 
341  $words = array_filter( $words );
342 
343  foreach ( $words as $word ) {
344  $search_criteria['field_filters'][] = array(
345  'key' => null, // The field ID to search
346  'value' => $word, // The value to search
347  'operator' => 'contains', // What to search in. Options: `is` or `contains`
348  );
349  }
350  }
351 
352  //start date & end date
353  $curr_start = esc_attr( rgget( 'gv_start' ) );
354  $curr_end = esc_attr( rgget( 'gv_end' ) );
355 
356  /**
357  * @filter `gravityview_date_created_adjust_timezone` Whether to adjust the timezone for entries. \n
358  * date_created is stored in UTC format. Convert search date into UTC (also used on templates/fields/date_created.php)
359  * @since 1.12
360  * @param[out,in] boolean $adjust_tz Use timezone-adjusted datetime? If true, adjusts date based on blog's timezone setting. If false, uses UTC setting. Default: true
361  * @param[in] string $context Where the filter is being called from. `search` in this case.
362  */
363  $adjust_tz = apply_filters( 'gravityview_date_created_adjust_timezone', true, 'search' );
364  $search_criteria['start_date'] = ( $adjust_tz && !empty( $curr_start ) ) ? get_gmt_from_date( $curr_start ) : $curr_start;
365  $search_criteria['end_date'] = ( $adjust_tz && !empty( $curr_end ) ) ? get_gmt_from_date( $curr_end ) : $curr_end;
366 
367 
368  // search for a specific entry ID
369  if ( ! empty( $_GET[ 'gv_id' ] ) ) {
370  $search_criteria['field_filters'][] = array(
371  'key' => 'id',
372  'value' => absint( $_GET[ 'gv_id' ] ),
373  'operator' => '=',
374  );
375  }
376 
377  // search for a specific Created_by ID
378  if ( ! empty( $_GET[ 'gv_by' ] ) ) {
379  $search_criteria['field_filters'][] = array(
380  'key' => 'created_by',
381  'value' => absint( $_GET['gv_by'] ),
382  'operator' => '=',
383  );
384  }
385 
386 
387  // search mode is 'ANY' by default
388  $mode = 'any';
389 
390  // get the other search filters
391  foreach ( $_GET as $key => $value ) {
392 
393  if ( 0 !== strpos( $key, 'filter_' ) || empty( $value ) || ( is_array( $value ) && count( $value ) === 1 && empty( $value[0] ) ) ) {
394  continue;
395  }
396 
397  // could return simple filter or multiple filters
398  $filter = $this->prepare_field_filter( $key, $value );
399 
400  if ( isset( $filter[0]['value'] ) ) {

402 
403  // if date range type, set search mode to ALL
404  if ( ! empty( $filter[0]['operator'] ) && in_array( $filter[0]['operator'], array( '>', '<' ) ) ) {
405  $mode = 'all';
406  }
407  } elseif( !empty( $filter ) ) {
408  $search_criteria['field_filters'][] = $filter;
409  }
410  }

414  * @since 1.5.1
415  * @param[out,in] string $mode Search mode (`any` vs `all`)
416  */
417  $search_criteria['field_filters']['mode'] = apply_filters( 'gravityview/search/mode', $mode );

419  do_action( 'gravityview_log_debug', sprintf( '%s[filter_entries] Returned Search Criteria: ', get_class( $this ) ), $search_criteria );

424  /**
425  * Prepare the field filters to GFAPI
426  *
427  * The type post_category, multiselect and checkbox support multi-select search - each value needs to be separated in an independent filter so we could apply the ANY search mode.
428  *
429  * Format searched values
430  * @param string $key $_GET search key
431  * @param string $value $_GET search value
432  * @return array 1 or 2 deph levels
433  */
434  public function prepare_field_filter( $key, $value ) {
435 
436  $gravityview_view = GravityView_View::getInstance();
437 
438  // calculates field_id, removing 'filter_' and for '_' for advanced fields ( like name or checkbox )
439  $field_id = str_replace( '_', '.', str_replace( 'filter_', '', $key ) );
440 
441  // get form field array
442  $form = $gravityview_view->getForm();
443  $form_field = gravityview_get_field( $form, $field_id );
444 
445  // default filter array
446  $filter = array(
447  'key' => $field_id,
448  'value' => $value,
449  );
450 
451  switch ( $form_field['type'] ) {
452 
453  case 'post_category':
454 
455  if ( ! is_array( $value ) ) {
456  $value = array( $value );
457  }
458 
459  // Reset filter variable
460  $filter = array();
461 
462  foreach ( $value as $val ) {
463  $cat = get_term( $val, 'category' );
464  $filter[] = array( 'key' => $field_id, 'value' => esc_attr( $cat->name ) . ':' . $val );
465  }
466 
467  break;
468 
469  case 'multiselect':
470 
471  if ( ! is_array( $value ) ) {
472  break;
473  }
474 
475  // Reset filter variable
476  $filter = array();
477 
478  foreach ( $value as $val ) {
479  $filter[] = array( 'key' => $field_id, 'value' => $val );
480  }
481 
482  break;
483 
484  case 'checkbox':
485  // convert checkbox on/off into the correct search filter
486  if ( false !== strpos( $field_id, '.' ) && ! empty( $form_field['inputs'] ) && ! empty( $form_field['choices'] ) ) {
487  foreach ( $form_field['inputs'] as $k => $input ) {
488  if ( $input['id'] == $field_id ) {
489  $filter['value'] = $form_field['choices'][ $k ]['value'];
490  break;
491  }
492  }
493  } elseif ( is_array( $value ) ) {
494 
495  // Reset filter variable
496  $filter = array();
497 
498  foreach ( $value as $val ) {
499  $filter[] = array( 'key' => $field_id, 'value' => $val );
500  }
501  }
502 
503  break;
504 
505  case 'name':
506  case 'address':
507 
508  if ( false === strpos( $field_id, '.' ) ) {
509 
510  $words = explode( ' ', $value );
511 
512  $filters = array();
513  foreach ( $words as $word ) {
514  if ( ! empty( $word ) && strlen( $word ) > 1 ) {
515  // Keep the same key for each filter
516  $filter['value'] = $word;
517  // Add a search for the value
518  $filters[] = $filter;
519  }
520  }
521 
522  $filter = $filters;
523  }
524 
525  break;
526 
527  case 'date':
528 
529  if ( is_array( $value ) ) {
530 
531  // Reset filter variable
532  $filter = array();
533 
534  foreach ( $value as $k => $date ) {
535  if ( empty( $date ) ) {
536  continue;
537  }
538  $operator = 'start' === $k ? '>' : '<';
539 
540  $filter[] = array(
541  'key' => $field_id,
542  'value' => self::get_formatted_date( $date, 'Y-m-d' ),
543  'operator' => $operator,
544  );
545  }
546  } else {
547  $filter['value'] = self::get_formatted_date( $value, 'Y-m-d' );
548  }
549 
550  break;
551 
552 
553  } // switch field type
554 
555  return $filter;
556  }
