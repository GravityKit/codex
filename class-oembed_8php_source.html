76 
77  $entry_var_name = GravityView_Post_Types::get_entry_var_name();
78 
79  /**
80  * @filter `gravityview_slug` Modify the url part for a View. [Read the doc](http://docs.gravityview.co/article/62-changing-the-view-slug)
81  * @param string $rewrite_slug The slug shown in the URL
82  */
83  $rewrite_slug = apply_filters( 'gravityview_slug', 'view' );
84 
85  // Only support embeds for current site
86  $prefix = trailingslashit( home_url() );
87 
88  // Using permalinks
89  $using_permalinks = $prefix . "(?P<is_cpt>{$rewrite_slug})?/?(?P<slug>.+?)/{$entry_var_name}/(?P<entry_slug>.+?)/?\$";
90 
91  // Not using permalinks
92  $not_using_permalinks = $prefix . "(?:index.php)?\?(?P<is_cpt2>[^=]+)=(?P<slug2>[^&]+)&entry=(?P<entry_slug2>[^&]+)\$";
93 
94  // Catch either
95  $match_regex = "(?:{$using_permalinks}|{$not_using_permalinks})";
96 
97  return '#'.$match_regex.'#i';

158 
159  // If not using permalinks, re-assign values for matching groups
160  if( !empty( $matches['entry_slug2'] ) ) {
161  $matches['is_cpt'] = $matches['is_cpt2'];
162  $matches['slug'] = $matches['slug2'];
163  $matches['entry_slug'] = $matches['entry_slug2'];
164  unset( $matches['is_cpt2'], $matches['slug2'], $matches['entry_slug2'] );
165  }
166 
167  // No Entry was found
168  if( empty( $matches['entry_slug'] ) ) {
169 
170  do_action('gravityview_log_error', 'GravityView_oEmbed[render_handler] $entry_slug not parsed by regex.', $matches );
171 
172  return '';
173  }
174 
175  $return = '';
176 
177  // Setup the data used
178  $this->set_vars( $matches, $attr, $url, $rawattr );
179 
180  if( is_admin() && !$this->is_full_oembed_preview ) {

185  $return .= $this->generate_preview_notice();

187 
188  $return .= $this->render_frontend( $matches, $attr, $url, $rawattr );
189  }

308  * @param array $attr Embed attributes.

310  * @param array $rawattr The original unmodified attributes.
311  * @return string The embed HTML.
312  */
313  private function render_frontend( $matches, $attr, $url, $rawattr ) {
314 
315  // If it's already been parsed, don't re-output it.
316  if( !empty( $this->output[ $this->entry_id ] ) ) {
317  return $this->output[ $this->entry_id ];
318  }
319 
320  $entry_output = $this->generate_entry_output();
321 
322  // Wrap a container div around the output to allow for custom styling
323  $output = sprintf('<div class="gravityview-oembed gravityview-oembed-entry gravityview-oembed-entry-'.$this->entry_id.'">%s</div>', $entry_output );
324 
325  /**
326  * @filter `gravityview/oembed/entry` Filter the output of the oEmbed entry embed
327  * @param string $output HTML of the embedded entry, with wrapper div
328  * @param GravityView_oEmbed $object The current GravityView_oEmbed instance
329  * @param array $atts Other passed parameters and info. \n
330  * @var string $entry_output HTML of just the View output, without the wrapper \n
331  * @var array $matches Capture group matches from the regex \n
332  * @var array $attr Embed attributes. \n
333  * @var string $url The original URL that was matched by the regex. \n
334  * @var array $rawattr The original unmodified attributes.
335  */
336  $output = apply_filters('gravityview/oembed/entry', $output, $this, compact( $entry_output, $matches, $attr, $url, $rawattr ) );
