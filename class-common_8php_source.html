412  'search_criteria' => null,
413  'sorting' => null,

416  );
417 
418  $criteria = wp_parse_args( $passed_criteria, $search_criteria_defaults );
419 
420  if ( ! empty( $criteria['search_criteria']['field_filters'] ) ) {
421  foreach ( $criteria['search_criteria']['field_filters'] as &$filter ) {
422 
423  if ( ! is_array( $filter ) ) {
424  continue;
425  }
426 
427  // By default, we want searches to be wildcard for each field.
428  $filter['operator'] = empty( $filter['operator'] ) ? 'contains' : $filter['operator'];
429 
430  /**
431  * @filter `gravityview_search_operator` Modify the search operator for the field (contains, is, isnot, etc)
432  * @param string $operator Existing search operator
433  * @param array $filter array with `key`, `value`, `operator`, `type` keys
434  */
435  $filter['operator'] = apply_filters( 'gravityview_search_operator', $filter['operator'], $filter );
436  }
437  }
438 
439  /**
440  * Prepare date formats to be in Gravity Forms DB format;
441  * $passed_criteria may include date formats incompatible with Gravity Forms.
442  */
443  foreach ( array('start_date', 'end_date' ) as $key ) {
444 
445  if ( ! empty( $criteria['search_criteria'][ $key ] ) ) {
446 
447  // Use date_create instead of new DateTime so it returns false if invalid date format.
448  $date = date_create( $criteria['search_criteria'][ $key ] );
449 
450  if ( $date ) {
451  // Gravity Forms wants dates in the `Y-m-d H:i:s` format.
452  $criteria['search_criteria'][ $key ] = $date->format( 'Y-m-d H:i:s' );
453  } else {
454  // If it's an invalid date, unset it. Gravity Forms freaks out otherwise.
455  unset( $criteria['search_criteria'][ $key ] );
456 
457  do_action( 'gravityview_log_error', '[filter_get_entries_criteria] '.$key.' Date format not valid:', $criteria['search_criteria'][ $key ] );
458  }
459  }
460  }

462

464  if ( class_exists( 'GravityView_View_Data' ) && GravityView_View_Data::getInstance()->has_multiple_views() || GravityView_frontend::getInstance()->single_entry ) {

569  /**

572  * Since 1.4, supports custom entry slugs. The way that GravityView fetches an entry based on the custom slug is by searching `gravityview_unique_id` meta. The `$entry_slug` is fetched by getting the current query var set by `is_single_entry()`

577  * @param boolean $check_entry_display Check whether the entry is visible for the current View configuration. Default: true. {@since 1.14}
578  * @return array|boolean

580  public static function get_entry( $entry_slug, $force_allow_ids = false, $check_entry_display = true ) {

589 
590  /**

592  * - If disabled (default), only allow access to an entry using the custom slug value. (example: `/entry/custom-slug/` NOT `/entry/123/`)
593  * - If enabled, you could access using the custom slug OR the entry id (example: `/entry/custom-slug/` OR `/entry/123/`)
594  * @param boolean $custom_slug_id_access True: allow accessing the slug by ID; False: only use the slug passed to the method.
595  */
596  $custom_slug_id_access = $force_allow_ids || apply_filters( 'gravityview_custom_entry_slug_allow_id', false );
597

600  * instead of doing a straightup ID search.
601  */
602  if ( $custom_slug ) {
603 
604  $entry_id = self::get_entry_id_from_slug( $entry_slug );
605 
606  }
607 
608  // If custom slug is off, search using the entry ID
609  // ID allow ID access is on, also use entry ID as a backup
610  if ( empty( $custom_slug ) || ! empty( $custom_slug_id_access ) ) {
611  // Search for IDs matching $entry_slug
612  $entry_id = $entry_slug;
613  }
614 
615  if ( empty( $entry_id ) ) {
616  return false;
617  }
618 
619  // fetch the entry
620  $entry = GFAPI::get_entry( $entry_id );
621 
622  if( $check_entry_display ) {
623  // Is the entry allowed
624  $entry = self::check_entry_display( $entry );
625  }
626 
627  return is_wp_error( $entry ) ? false : $entry;
628 
629  }
630 
631  return false;
632  }

641  * @see http://docs.gravityview.co/article/252-gvlogic-shortcode
642  * @uses GFFormsModel::matches_operation
643  * @since 1.7.5
644  *
645  * @param string $val1 Left side of comparison
646  * @param string $val2 Right side of comparison
647  * @param string $operation Type of comparison
648  *
649  * @return bool True: matches, false: not matches
650  */
651  public static function matches_operation( $val1, $val2, $operation ) {
652 
653  $value = false;
654 
655  if( 'context' === $val1 ) {
656 
657  $matching_contexts = array( $val2 );
658 
659  // We allow for non-standard contexts.
660  switch( $val2 ) {
661  // Check for either single or edit
662  case 'singular':
663  $matching_contexts = array( 'single', 'edit' );
664  break;
665  // Use multiple as alias for directory for consistency
666  case 'multiple':
667  $matching_contexts = array( 'directory' );
668  break;
669  }
670 
671  $val1 = in_array( gravityview_get_context(), $matching_contexts ) ? $val2 : false;
672  }
673 
674  switch ( $operation ) {
675  case 'equals':
676  $value = GFFormsModel::matches_operation( $val1, $val2, 'is' );
677  break;
678  case 'greater_than_or_is':
679  case 'greater_than_or_equals':
680  $is = GFFormsModel::matches_operation( $val1, $val2, 'is' );
681  $gt = GFFormsModel::matches_operation( $val1, $val2, 'greater_than' );
682  $value = ( $is || $gt );
683  break;
684  case 'less_than_or_is':
685  case 'less_than_or_equals':
686  $is = GFFormsModel::matches_operation( $val1, $val2, 'is' );
687  $gt = GFFormsModel::matches_operation( $val1, $val2, 'less_than' );
688  $value = ( $is || $gt );
689  break;
690  case 'not_contains':
691  $contains = GFFormsModel::matches_operation( $val1, $val2, 'contains' );
692  $value = ! $contains;
693  break;
694  default:
695  $value = GFFormsModel::matches_operation( $val1, $val2, $operation );
696  }
697 
698  return $value;
699  }

716  return false;

747  }
748 
749  $filters = $search_criteria['field_filters'];
750  unset( $search_criteria );

752  $mode = array_key_exists( 'mode', $filters ) ? strtolower( $filters['mode'] ) : 'all';
753  unset( $filters['mode'] );

782  }
783  }

786  // Or, for ANY mode, means none of the conditions were satisfied, so entry is not approved
787  if ( 'all' === $mode ) {
788  return $entry;
789  } else {
