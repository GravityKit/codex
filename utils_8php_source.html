407  * Determine approximate backtrack count when running PCRE.
408  *
409  * @return int The backtrack count.
410  */
411 function benchmark_pcre_backtracking( $pattern, $subject, $strategy ) {
412  $saved_config = ini_get( 'pcre.backtrack_limit' );
413 
414  // Attempt to prevent PHP crashes. Adjust these lower when needed.
415  if ( version_compare( phpversion(), '5.4.8', '>' ) ) {
416  $limit = 1000000;
417  } else {
418  $limit = 20000; // 20,000 is a reasonable upper limit, but see also https://core.trac.wordpress.org/ticket/29557#comment:10
419  }
420 
421  // Start with small numbers, so if a crash is encountered at higher numbers we can still debug the problem.
422  for( $i = 4; $i <= $limit; $i *= 2 ) {
423 
424  ini_set( 'pcre.backtrack_limit', $i );
425 
426  switch( $strategy ) {
427  case 'split':
428  preg_split( $pattern, $subject );
429  break;
430  case 'match':
431  preg_match( $pattern, $subject );
432  break;
433  case 'match_all':
434  $matches = array();
435  preg_match_all( $pattern, $subject, $matches );
436  break;
437  }
438 
439  ini_set( 'pcre.backtrack_limit', $saved_config );
440 
441  switch( preg_last_error() ) {
442  case PREG_NO_ERROR:
443  return $i;
444  case PREG_BACKTRACK_LIMIT_ERROR:
445  continue;
446  case PREG_RECURSION_LIMIT_ERROR:
447  trigger_error('PCRE recursion limit encountered before backtrack limit.');
448  return;
449  case PREG_BAD_UTF8_ERROR:
450  trigger_error('UTF-8 error during PCRE benchmark.');
451  return;
452  case PREG_INTERNAL_ERROR:
453  trigger_error('Internal error during PCRE benchmark.');
454  return;
455  default:
456  trigger_error('Unexpected error during PCRE benchmark.');
457  return;
458  }
459  }
460 
461  return $i;
462 }
