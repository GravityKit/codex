765  *
766  * The usage of the get_term function is to apply filters to a term object. It
767  * is possible to get a term object from the database before applying the
768  * filters.

771  * be able to be captured by the hooks. Failure would be the same value as $wpdb

798  * available. If stdClass object (as in the results of a database query), will apply

876  * required.

879  * field, but not recommended that you do so.

882  * and $field and $value combinations exist, the Term will be returned.

891  * @see sanitize_term_field() The $context param lists the available values for get_term_by() $filter param.

895  * @param string $taxonomy Taxonomy name. Optional, if `$field` is 'term_taxonomy_id'.
896  * @param string $output Constant OBJECT, ARRAY_A, or ARRAY_N

902  global $wpdb;

910

912  $_field = 't.slug';
913  $value = sanitize_title($value);
914  if ( empty($value) )
915  return false;
916  } elseif ( 'name' == $field ) {
917  // Assume already escaped
918  $value = wp_unslash($value);
919  $_field = 't.name';
920  } elseif ( 'term_taxonomy_id' == $field ) {
921  $value = (int) $value;
922  $_field = 'tt.term_taxonomy_id';
923 
924  // No `taxonomy` clause when searching by 'term_taxonomy_id'.
925  $tax_clause = '';
926  } else {
927  $term = get_term( (int) $value, $taxonomy, $output, $filter );
928  if ( is_wp_error( $term ) || is_null( $term ) ) {
929  $term = false;
930  }

932  }
933

941  }
942 
943  wp_cache_add( $term->term_id, $term, 'terms' );
