641  * @since 3.4.0
642  * @access private
643  *
644  * @staticvar array $header_tags
645  * @staticvar array $header_tags_with_a
646  *
647  * @param string $header Theme header. Name, Description, Author, Version, ThemeURI, AuthorURI, Status, Tags.
648  * @param string $value Value to sanitize.
649  * @return mixed
650  */
651  private function sanitize_header( $header, $value ) {
652  switch ( $header ) {
653  case 'Status' :
654  if ( ! $value ) {
655  $value = 'publish';
656  break;
657  }
658  // Fall through otherwise.
659  case 'Name' :
660  static $header_tags = array(
661  'abbr' => array( 'title' => true ),
662  'acronym' => array( 'title' => true ),
663  'code' => true,
664  'em' => true,
665  'strong' => true,
666  );
667  $value = wp_kses( $value, $header_tags );
668  break;
669  case 'Author' :
670  // There shouldn't be anchor tags in Author, but some themes like to be challenging.
671  case 'Description' :
672  static $header_tags_with_a = array(
673  'a' => array( 'href' => true, 'title' => true ),
674  'abbr' => array( 'title' => true ),
675  'acronym' => array( 'title' => true ),
676  'code' => true,
677  'em' => true,
678  'strong' => true,
679  );
680  $value = wp_kses( $value, $header_tags_with_a );
681  break;
682  case 'ThemeURI' :
683  case 'AuthorURI' :
684  $value = esc_url_raw( $value );
685  break;
686  case 'Tags' :
687  $value = array_filter( array_map( 'trim', explode( ',', strip_tags( $value ) ) ) );
688  break;
689  case 'Version' :
690  $value = strip_tags( $value );
691  break;
692  }
693 
694  return $value;
695  }

697  /**
698  * Mark up a theme header.
699  *
700  * @since 3.4.0
701  * @access private
702  *
703  * @staticvar string $comma
704  *
705  * @param string $header Theme header. Name, Description, Author, Version, ThemeURI, AuthorURI, Status, Tags.
706  * @param string $value Value to mark up.
707  * @param string $translate Whether the header has been translated.
708  * @return string Value, marked up.
709  */
710  private function markup_header( $header, $value, $translate ) {
711  switch ( $header ) {
712  case 'Name' :
713  if ( empty( $value ) )
714  $value = $this->get_stylesheet();
715  break;
716  case 'Description' :
717  $value = wptexturize( $value );
718  break;
719  case 'Author' :
720  if ( $this->get('AuthorURI') ) {
721  $value = sprintf( '<a href="%1$s">%2$s</a>', $this->display( 'AuthorURI', true, $translate ), $value );
722  } elseif ( ! $value ) {
723  $value = __( 'Anonymous' );
724  }
725  break;
726  case 'Tags' :
727  static $comma = null;
728  if ( ! isset( $comma ) ) {
729  /* translators: used between list items, there is a space after the comma */
730  $comma = __( ', ' );
731  }
732  $value = implode( $comma, $value );
733  break;
734  case 'ThemeURI' :
735  case 'AuthorURI' :
736  $value = esc_url( $value );
737  break;
738  }
739 
740  return $value;
741  }

744  * Translate a theme header.
745  *
746  * @since 3.4.0
747  * @access private
748  *
749  * @staticvar array $tags_list
750  *
751  * @param string $header Theme header. Name, Description, Author, Version, ThemeURI, AuthorURI, Status, Tags.
752  * @param string $value Value to translate.
753  * @return string Translated value.
754  */
755  private function translate_header( $header, $value ) {
756  switch ( $header ) {
757  case 'Name' :
758  // Cached for sorting reasons.
759  if ( isset( $this->name_translated ) )
760  return $this->name_translated;
761  $this->name_translated = translate( $value, $this->get('TextDomain' ) );
762  return $this->name_translated;
763  case 'Tags' :
764  if ( empty( $value ) || ! function_exists( 'get_theme_feature_list' ) )
765  return $value;

767  static $tags_list;
768  if ( ! isset( $tags_list ) ) {
769  $tags_list = array();
770  $feature_list = get_theme_feature_list( false ); // No API
771  foreach ( $feature_list as $tags )
772  $tags_list += $tags;

775  foreach ( $value as &$tag ) {
776  if ( isset( $tags_list[ $tag ] ) ) {
777  $tag = $tags_list[ $tag ];

779  $tag = $tags_list[ self::$tag_map[ $tag ] ];
780  }
781  }
782 
783  return $value;
784 
785  default :
786  $value = translate( $value, $this->get('TextDomain') );

1037  /**
1038  * Scans a directory for files of a certain extension.
1039  *
1040  * @since 3.4.0
1041  *
1042  * @static
1043  * @access private
1044  *
1045  * @param string $path Absolute path to search.
1046  * @param array|string|null $extensions Optional. Array of extensions to find, string of a single extension,
1047  * or null for all extensions. Default null.
1048  * @param int $depth Optional. How many levels deep to search for files. Accepts 0, 1+, or
1049  * -1 (infinite depth). Default 0.
1050  * @param string $relative_path Optional. The basename of the absolute path. Used to control the
1051  * returned path for the found files, particularly when this function
1052  * recurses to lower depths. Default empty.
1053  * @return array|false Array of files, keyed by the path to the file relative to the `$path` directory prepended
1054  * with `$relative_path`, with the values being absolute paths. False otherwise.
1055  */
1056  private static function scandir( $path, $extensions = null, $depth = 0, $relative_path = '' ) {
1057  if ( ! is_dir( $path ) )
1058  return false;
1059 
1060  if ( $extensions ) {
1061  $extensions = (array) $extensions;
1062  $_extensions = implode( '|', $extensions );
1063  }
1064 
1065  $relative_path = trailingslashit( $relative_path );
1066  if ( '/' == $relative_path )
1067  $relative_path = '';
1068 
1069  $results = scandir( $path );
1070  $files = array();
1071 
1072  foreach ( $results as $result ) {
1073  if ( '.' == $result[0] )
1074  continue;
1075  if ( is_dir( $path . '/' . $result ) ) {
1076  if ( ! $depth || 'CVS' == $result )
1077  continue;
1078  $found = self::scandir( $path . '/' . $result, $extensions, $depth - 1 , $relative_path . $result );
1079  $files = array_merge_recursive( $files, $found );
1080  } elseif ( ! $extensions || preg_match( '~\.(' . $_extensions . ')$~', $result ) ) {
1081  $files[ $relative_path . $result ] = $path . '/' . $result;
1082  }
1083  }
1084 
1085  return $files;
1086  }
