605  * a 404 so that canonical redirection logic can kick in.
606  *
607  * @since 2.0.0
608  *
609  * @global WP_Query $wp_query
610  */
611  public function handle_404() {
612  global $wp_query;
613 
614  // If we've already issued a 404, bail.
615  if ( is_404() )
616  return;
617 
618  // Never 404 for the admin, robots, or if we found posts.
619  if ( is_admin() || is_robots() || $wp_query->posts ) {
620 
621  $success = true;
622  if ( is_singular() ) {
623  $p = false;
624 
625  if ( $wp_query->post instanceof WP_Post ) {
626  $p = clone $wp_query->post;
627  }
628 
629  // Only set X-Pingback for single posts that allow pings.
630  if ( $p && pings_open( $p ) ) {
631  @header( 'X-Pingback: ' . get_bloginfo( 'pingback_url' ) );
632  }
633 
634  // check for paged content that exceeds the max number of pages
635  $next = '<!--nextpage-->';
636  if ( $p && false !== strpos( $p->post_content, $next ) && ! empty( $this->query_vars['page'] ) ) {
637  $page = trim( $this->query_vars['page'], '/' );
638  $success = (int) $page <= ( substr_count( $p->post_content, $next ) + 1 );
639  }
640  }
641 
642  if ( $success ) {
643  status_header( 200 );
644  return;
645  }
646  }
647 
648  // We will 404 for paged queries, as no posts were found.
649  if ( ! is_paged() ) {
650 
651  // Don't 404 for authors without posts as long as they matched an author on this site.
652  $author = get_query_var( 'author' );
653  if ( is_author() && is_numeric( $author ) && $author > 0 && is_user_member_of_blog( $author ) ) {
654  status_header( 200 );
655  return;
656  }
657 
658  // Don't 404 for these queries if they matched an object.
659  if ( ( is_tag() || is_category() || is_tax() || is_post_type_archive() ) && get_queried_object() ) {
660  status_header( 200 );
661  return;
662  }
663 
664  // Don't 404 for these queries either.
665  if ( is_home() || is_search() || is_feed() ) {
666  status_header( 200 );
667  return;
668  }
669  }
670 
671  // Guess it's time to 404.
672  $wp_query->set_404();
673  status_header( 404 );
674  nocache_headers();
675  }
